    $( ".choice" ).on( "click", function(e) {
        e.preventDefault();
        var href = $(this).attr('href');
        var parms = window.location.search;
        if (parms != null || parms.length > 0 || parms.charAt(0) == '?') {
            parms = parms.substring(1);
        }
        else
        {
            parms = '';
        }
        window.location = href + '?' + parms;
        });
});
